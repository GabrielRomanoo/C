10. Qual a saída do programa abaixo?
#include <stdio.h>
int x;
void fun(void);
int main() {
 printf("main: x = %d\n", x++); //o ++ so é incrementado depois do printf
 fun();
 printf("main: x = %d\n", x++);
 fun();
 printf("main: x = %d\n", x++);
 return 0;
}
void fun() {
 static int y;  //continua mantendo seu valor mesmo encerrando a funcao
 printf("fun: x = %d, y = %d\n", x, y++);
 auto int x = y * 2;                       <- passa por cima do x global
 printf("fun: x = %d, y = %d\n", x, y++);
}

SAÍDA: 
main : x = 0    /dps que é impresso, é incrementado 1 em x global
fun: x = 1, y = 0  //dps que é impresso, é incrementado 1 em y static
fun: x = 2, y = 1  <- esse x é diferente do x global
main: x = 1
fun: x = 2, y = 2
fun: x = 6, y = 3
main: x = 2
