para a função abaixo.
int menor_pos(int n, int v[]) 
{
   int i, pos = 0;
   for (i = 1; i < n; i++) {
      if (v[i] < v[pos])
         pos = i;      
   }
   return pos;
}

Resposta:

/*
A funcao recebe recebe um numero n>=1 e um vetor com n elementos.
A funcao devolve a posicao do menor numero dos elementos de v.
*/

int menor_pos(int n, int v[]) 
{
   int i, pos = 0;
   for (i = 1; i < n; i++) {
/* pos é a posicao do menor elemento de v[0...i-1] */
      if (v[i] < v[pos])
         pos = i;      
   }
   return pos;
}
